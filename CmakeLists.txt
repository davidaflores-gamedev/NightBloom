#------------------------------------------------------------------------------
# NightBloom_Engine/CMakeLists.txt (ROOT)
#
# Root build configuration - builds both Engine and Editor
#------------------------------------------------------------------------------
cmake_minimum_required(VERSION 3.20)
project(NightbloomWorkspace VERSION 0.1.0 LANGUAGES CXX)

#------------------------------------------------------------------------------
# Global C++ settings (applies to ALL subprojects)
#------------------------------------------------------------------------------
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

#------------------------------------------------------------------------------
# Build Engine (includes ThirdParty automatically)
#------------------------------------------------------------------------------
add_subdirectory(NightBloom)

#------------------------------------------------------------------------------
# Build Editor (links against engine)
#------------------------------------------------------------------------------
add_subdirectory(Editor)

#------------------------------------------------------------------------------
# Set Editor as the startup project in Visual Studio
#------------------------------------------------------------------------------
set_property(DIRECTORY ${CMAKE_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT NightbloomEditor)
